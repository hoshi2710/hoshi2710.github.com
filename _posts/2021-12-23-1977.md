#include <stdio.h>

int main() {
    int range[2];
    for (int i=0; i<2; i++)
        scanf("%d",&range[i]);
    int i=1,min = 0,tot = 0;
    while(i*i <= range[1])
    {
        if(i*i>=range[0])
        {
            tot += i*i;
            if(min == 0)
                min = i*i;
        }
        i++;
    }
    if (tot == 0)
        printf("-1\n");
    else
        printf("%d\n%d\n",tot,min);
    return 0;
}
