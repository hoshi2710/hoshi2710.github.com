#include <stdio.h>

int main() {
    int odd_total=0,odd_min = 2147483647;
    for (int i=0; i<7; i++)
    {
        int temp;
        scanf("%d",&temp);
        if (temp % 2 == 1)
        {
            odd_total += temp;
            if(odd_min > temp)
                odd_min = temp;
        }
    }
    if (odd_total == 0 && odd_min == 2147483647)
        printf("-1\n");
    else
        printf("%d\n%d\n",odd_total,odd_min);
    return 0;
}
