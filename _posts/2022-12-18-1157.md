#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>

int main()
{
    char inp = '\0';
    int count[26] = { 0 },max=0,finded=0,chr=0;
    while (inp != '\n')
    {
        inp = getchar();
        count[inp - (inp >= 'a' && inp <= 'z' ? 'a' : 'A')] += 1;
    }
    for (int i = 0; i < 26; i++)
        if(count[i] != 0)
            if (max < count[i])
            {
                max = count[i];
                chr = i + 'A';
            }

    for (int i = 0; i < 26; i++)
    {
        finded += max == count[i] ? 1 : 0;
        if (finded > 1)
            break;
    }
    printf("%c", finded > 1 ? '?' : chr);
}
