#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main()
{
	int month_first_yoil[12] = {0,3,3,6,1,4,6,2,5,0,3,5},month=0,day=0;
	scanf("%d %d", &month, &day);
	int yoil = month_first_yoil[month - 1];
	yoil += day;
	if (yoil % 7 != 0)
		yoil %= 7;
	else
		yoil = 7;
	yoil -= 1;
	switch (yoil)
	{
		case 0:
			printf("MON");
			break;
		case 1:
			printf("TUE");
			break;
		case 2:
			printf("WED");
			break;
		case 3:
			printf("THU");
			break;
		case 4:
			printf("FRI");
			break;
		case 5:
			printf("SAT");
			break;
		case 6:
			printf("SUN");
			break;
		default:
			break;
	}

	return 0;
}
