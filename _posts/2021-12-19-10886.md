#include <stdio.h>

int main() {
    int n,cute=0,n_cute=0;
    scanf("%d",&n);
    for (int i=0; i<n;i++)
    {
        int temp;
        scanf("%d",&temp);
        if (temp == 1)
            cute++;
        else if (temp == 0)
            n_cute++;
    }
    if (cute > n_cute)
        printf("Junhee is cute!\n");
    else
        printf("Junhee is not cute!\n");
    
    
    return 0;
}
