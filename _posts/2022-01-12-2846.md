#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int start = 0, end = 0, temp_start, temp_end;
void save()
{
	if (end - start < temp_end - temp_start)
	{
		start = temp_start;
		end = temp_end;
	}
}
int main()
{
	int n, array[1000], increase = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		scanf("%d", &array[i]);
		if (i != 0)
		{
			if (array[i - 1] < array[i])
			{
				if(increase  == 0)
					temp_start = array[i - 1];
				if (i == n-1)
				{
					temp_end = array[i];
					save();
				}
				increase++;
			}
			else if (array[i - 1] >= array[i])
			{
				if (increase != 0)
				{
					if (i != n-1)
						temp_end = array[i - 1];
					else
						temp_end = array[i];
					save();
					increase = 0;	
				}
			}
			
		}
	}
	printf("%d", end-start);
	return 0;
}
