#include <stdio.h>

int main() {
    int n;
    scanf("%d",&n);
    for(int i=0; i<n; i++)
    {
        int n_benefit, y_benefit, adv_cost;
        scanf("%d %d %d",&n_benefit, &y_benefit, &adv_cost);
        if(n_benefit<y_benefit-adv_cost)
            printf("advertise");
        else if (n_benefit == y_benefit-adv_cost)
            printf("does not matter");
        else if (n_benefit > y_benefit-adv_cost)
            printf("do not advertise");
        printf("\n");
    }
    return 0;
}
